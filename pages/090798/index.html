# Day 5-7：基础运算符（重点！）

学习内容：

  • 算术运算符：+, -, *, /, //, %, **
  • 赋值运算符：=, +=, -=, *=, /=
  • 比较运算符：==, !=, >, <, >=, <=

案例讲解：

# 算术运算符
a = 10
b = 3
print(f"{a} + {b} = {a + b}")    # 13
print(f"{a} / {b} = {a / b}")    # 3.333...
print(f"{a} // {b} = {a // b}")  # 3 (整除)
print(f"{a} % {b} = {a % b}")    # 1 (取余)
print(f"{a} ** {b} = {a ** b}")  # 1000 (幂运算)

# 赋值运算符
x = 5
x += 3    # 等价于 x = x + 3
print(f"x = {x}")  # 8

# 比较运算符
age = 18
print(f"年龄大于等于18岁吗? {age >= 18}")  # True
print(f"年龄等于20岁吗? {age == 20}")      # False

# Day 10-12：循环语句

学习内容：

  • for 循环
  • while 循环

案例讲解：

# for循环 - 打印乘法表
print("乘法表:")
for i in range(1, 6):  # 1到5
    for j in range(1, i + 1):
        print(f"{j}×{i}={i*j}", end="\t")
    print()  # 换行

# while循环 - 猜数字游戏
import random
secret_number = random.randint(1, 10)
guesses = 0

print("\n猜数字游戏(1-10):")
while guesses < 3:
    guess = int(input("猜一个数字: "))
    guesses += 1

    if guess == secret_number:
        print("恭喜你猜对了!")
        break
    elif guess < secret_number:
        print("太小了!")
    else:
        print("太大了!")
else:
    print(f"游戏结束! 正确答案是: {secret_number}")

# 日记本程序
def write_diary():
    """写日记功能"""
    print("\n=== 写日记 ===")
    date = datetime.now().strftime("%Y-%m-%d")
    content = input("请输入日记内容: ")

    with open("diary.txt", "a", encoding="utf-8") as file:
        file.write(f"{date}: {content}\n")
    print("日记保存成功!")

def read_diary():
    """读日记功能"""
    print("\n=== 读日记 ===")
    try:
        with open("diary.txt", "r", encoding="utf-8") as file:
            entries = file.readlines()

        if not entries:
            print("日记本为空")
            return

        print("所有日记条目:")
        for i, entry in enumerate(entries, 1):
            print(f"{i}. {entry.strip()}")

    except FileNotFoundError:
        print("日记文件不存在，请先写日记")

# 主程序
while True:
    print("\n=== 个人日记本 ===")
    print("1. 写日记")
    print("2. 读日记")
    print("3. 退出")

    choice = input("请选择功能(1-3): ")

    if choice == "1":
        write_diary()
    elif choice == "2":
        read_diary()
    elif choice == "3":
        print("感谢使用日记本，再见!")
        break
    else:
        print("无效选择，请重新输入")

# Day 24：异常处理

学习内容：

  • try, except, finally 语句
  • 常见异常类型

案例讲解：

# 银行账户系统
class BankAccount:
    def __init__(self, account_number, balance=0):
        self.account_number = account_number
        self.balance = balance

    def deposit(self, amount):
        """存款"""
        if amount <= 0:
            raise ValueError("存款金额必须大于0")
        self.balance += amount
        return self.balance

    def withdraw(self, amount):
        """取款"""
        if amount <= 0:
            raise ValueError("取款金额必须大于0")
        if amount > self.balance:
            raise ValueError("余额不足")
        self.balance -= amount
        return self.balance

# 使用异常处理
def main():
    account = BankAccount("123456", 1000)
    print(f"初始余额: {account.balance}元")

    while True:
        print("\n=== 银行系统 ===")
        print("1. 存款")
        print("2. 取款")
        print("3. 查询余额")
        print("4. 退出")

        try:
            choice = input("请选择功能(1-4): ")

            if choice == "1":
                amount = float(input("请输入存款金额: "))
                new_balance = account.deposit(amount)
                print(f"存款成功! 新余额: {new_balance}元")

            elif choice == "2":
                amount = float(input("请输入取款金额: "))
                new_balance = account.withdraw(amount)
                print(f"取款成功! 新余额: {new_balance}元")

            elif choice == "3":
                print(f"当前余额: {account.balance}元")

            elif choice == "4":
                print("感谢使用银行系统，再见!")
                break

            else:
                print("无效选择，请重新输入")

        except ValueError as e:
            print(f"错误: {e}")
        except Exception as e:
            print(f"系统错误: {e}")

if __name__ == "__main__":
    main()

# 图书管理系统
class Book:
    def __init__(self, title, author, isbn, available=True):
        self.title = title
        self.author = author
        self.isbn = isbn
        self.available = available

    def __str__(self):
        status = "可借" if self.available else "已借出"
        return f"《{self.title}》- {self.author} ({self.isbn}) [{status}]"

    def borrow(self):
        if not self.available:
            raise Exception("该书已被借出")
        self.available = False
        return True

    def return_book(self):
        if self.available:
            raise Exception("该书已在馆内")
        self.available = True
        return True

class Library:
    def __init__(self):
        self.books = []

    def add_book(self, book):
        self.books.append(book)
        print(f"添加图书: {book.title}")

    def find_book(self, keyword):
        results = []
        for book in self.books:
            if keyword.lower() in book.title.lower() or keyword.lower() in book.author.lower():
                results.append(book)
        return results

    def display_all_books(self):
        if not self.books:
            print("图书馆暂无图书")
            return

        print("\n=== 图书馆所有图书 ===")
        for i, book in enumerate(self.books, 1):
            print(f"{i}. {book}")

# 使用示例
library = Library()

# 添加图书
book1 = Book("Python编程从入门到实践", "Eric Matthes", "978-7-111-60755-1")
book2 = Book("算法导论", "Thomas H. Cormen", "978-7-111-40701-0")
book3 = Book("深入理解计算机系统", "Randal E. Bryant", "978-7-111-44348-8")

library.add_book(book1)
library.add_book(book2)
library.add_book(book3)

# 显示所有图书
library.display_all_books()

# 搜索图书
search_results = library.find_book("Python")
print("\n=== 搜索结果 ===")
for book in search_results:
    print(book)

# 借书
print("\n=== 借书操作 ===")
try:
    book1.borrow()
    print(f"成功借出: {book1.title}")
    library.display_all_books()
except Exception as e:
    print(f"借书失败: {e}")

# Day 27-30：综合项目实战

项目：天气查询系统

import requests
import json
from datetime import datetime

class WeatherApp:
    def __init__(self):
        self.api_key = "YOUR_API_KEY"  # 需要注册免费API
        self.base_url = "http://api.weatherapi.com/v1"

    def get_weather(self, city):
        """获取天气数据"""
        try:
            url = f"{self.base_url}/current.json"
            params = {
                'key': self.api_key,
                'q': city,
                'lang': 'zh'  # 中文
            }

            response = requests.get(url, params=params)
            response.raise_for_status()  # 检查HTTP错误

            data = response.json()
            return self._parse_weather_data(data)

        except requests.exceptions.RequestException as e:
            print(f"网络请求错误: {e}")
            return None
        except KeyError as e:
            print(f"数据解析错误: {e}")
            return None

    def _parse_weather_data(self, data):
        """解析天气数据"""
        current = data['current']
        location = data['location']

        weather_info = {
            'city': location['name'],
            'country': location['country'],
            'temp_c': current['temp_c'],
            'temp_f': current['temp_f'],
            'condition': current['condition']['text'],
            'humidity': current['humidity'],
            'wind_kph': current['wind_kph'],
            'wind_dir': current['wind_dir'],
            'last_updated': current['last_updated']
        }
        return weather_info

    def display_weather(self, weather_info):
        """显示天气信息"""
        if not weather_info:
            print("无法获取天气信息")
            return

        print(f"\n=== {weather_info['city']}, {weather_info['country']} 天气 ===")
        print(f"温度: {weather_info['temp_c']}°C ({weather_info['temp_f']}°F)")
        print(f"天气状况: {weather_info['condition']}")
        print(f"湿度: {weather_info['humidity']}%")
        print(f"风速: {weather_info['wind_kph']} km/h, 风向: {weather_info['wind_dir']}")
        print(f"更新时间: {weather_info['last_updated']}")

    def run(self):
        """运行天气应用"""
        print("=== 天气查询系统 ===")
        print("输入城市名称查询天气，输入'exit'退出")

        while True:
            city = input("\n请输入城市名称: ").strip()

            if city.lower() == 'exit':
                print("感谢使用天气查询系统，再见!")
                break

            if not city:
                print("请输入有效的城市名称")
                continue

            print(f"正在查询 {city} 的天气...")
            weather_info = self.get_weather(city)

            if weather_info:
                self.display_weather(weather_info)
            else:
                print(f"无法获取 {city} 的天气信息，请检查城市名称是否正确")

# 注意：实际使用需要注册免费的WeatherAPI
# 替换 YOUR_API_KEY 为你的API密钥
if __name__ == "__main__":
    app = WeatherApp()
    app.run()

# 🎯 学习建议

  1. 循序渐进：严格按照计划表学习，不要跳过基础内容
  2. 动手实践：每个案例都要亲自敲代码，不要复制粘贴
  3. 理解优先：先理解概念，再记忆语法
  4. 调试能力：学会使用Python 3.13改进的错误消息来调试代码
  5. 项目驱动：学完每个阶段后，尝试修改案例或创建小项目

# 🚀 Python 3.13 新特性利用

  • 新REPL：利用多行编辑功能来测试代码片段
  • 改进的错误消息：当代码出错时，Python 3.13会提供更清晰的错误提示
  • 虚拟环境改进：项目开发时使用虚拟环境隔离依赖

记住：编程是实践性很强的技能，每天坚持写代码比一次性学习大量理论更重要！遇到问题时，善用Python 3.13的新特性来帮助你更好地理解和调试代码。
